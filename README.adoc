= 说明
项目依赖于官方spring-boot, 在pom.xml文件里面配置
[source,java,indent=0]
----
<parent>
	<groupId>org.springframework.boot</groupId>
	<artifactId>spring-boot-starter-parent</artifactId>
	<version>1.1.0.RC1</version>
</parent>
----
